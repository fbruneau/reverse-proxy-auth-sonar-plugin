1   package net.trajano.sonar.plugins.reverseproxyauth;
2   
3   import org.sonar.api.config.Settings;
4   import org.sonar.api.security.ExternalUsersProvider;
5   import org.sonar.api.security.UserDetails;
6   
7   /**
8    * This provides the user details. It is called before
9    * {@link org.sonar.api.security.Authenticator#doAuthenticate(org.sonar.api.security.Authenticator.Context)}
10   * .
11   */
12  public class ReverseProxyAuthUsersProvider extends ExternalUsersProvider {
13  
14      /**
15       * Settings.
16       */
17      private final ReverseProxyAuthSettings settings;
18  
19      /**
20       * Constructs the {@link ExternalUsersProvider} with the specified
21       * {@link Settings}.
22       *
23       * @param settings
24       *            injected settings
25       */
26      public ReverseProxyAuthUsersProvider(final ReverseProxyAuthSettings settings) {
27          super();
28          this.settings = settings;
29      }
30  
31      /**
32       * Obtains the user details from the header. Will return <code>null</code>
33       * if it cannot be obtained. However, if it is localhost it will return an
34       * empty {@link UserDetails} structure. {@inheritDoc}
35       */
36      @Override
37      public UserDetails doGetUserDetails(final Context context) {
38  
39          final UserDetails userDetails = new UserDetails();
40          if (!settings.isLocalHost(context.getRequest())) {
41              final String headerValue = settings.getUserNameFromHeader(context.getRequest());
42              if (headerValue == null) {
43                  return null;
44              }
45              userDetails.setEmail(headerValue);
46              userDetails.setName(headerValue);
47          }
48          return userDetails;
49      }
50  }
