1   package net.trajano.sonar.plugins.reverseproxyauth;
2   
3   import javax.annotation.Nonnull;
4   import javax.servlet.ServletRequest;
5   import javax.servlet.http.HttpServletRequest;
6   
7   import org.sonar.api.CoreProperties;
8   import org.sonar.api.config.Settings;
9   import org.sonar.api.server.ServerSide;
10  
11  /**
12   * Wraps the settings object so it provides a type safe interface rather than
13   * using strings. This class also provides the property key constants.
14   *
15   * @author Archimedes
16   */
17  @ServerSide
18  public class ReverseProxyAuthSettings {
19  
20      /**
21       * If true, then new users that are not yet registered in the system are
22       * automatically registered.
23       */
24      public static final String ALLOW_NEW_USERS = "reverseproxyauth.allow.new";
25  
26      /**
27       * HTTP Header name containing the user name.
28       */
29      public static final String HEADER_NAME = "reverseproxyauth.header.name";
30  
31      /**
32       * Host name to allow sonar executions. Authentication will always be
33       * accepted when accessing this host name from the
34       * {@link javax.servlet.http.HttpServletRequest}.
35       */
36      public static final String LOCALHOST = "reverseproxyauth.localhost";
37  
38      /**
39       * Wrapped settings.
40       */
41      private final Settings settings;
42  
43      /**
44       * @param settings
45       *            injected settings.
46       */
47      public ReverseProxyAuthSettings(@Nonnull final Settings settings) {
48          this.settings = settings;
49      }
50  
51      /**
52       * Returns the value of {@link #ALLOW_NEW_USERS}.
53       *
54       * @return the value of {@link #ALLOW_NEW_USERS}.
55       */
56      public boolean allowsUsersToSignUp() {
57  
58          return settings.getBoolean(ALLOW_NEW_USERS);
59      }
60  
61      /**
62       * Base URL of the server.
63       *
64       * @return Base URL of the server.
65       */
66      public String getBaseUrl() {
67  
68          return settings.getString(CoreProperties.SERVER_BASE_URL);
69      }
70  
71      /**
72       * @return URL that points to the redirect back or home controller.
73       */
74      public String getRedirectBackOrHomeUrl() {
75  
76          return settings.getString(CoreProperties.SERVER_BASE_URL) + "/reverseproxyauth/redirect_back_or_home_url";
77      }
78  
79      /**
80       * This controller tells the Rails system to authenticate the user.
81       *
82       * @return URL that points to the reverseproxyauth init controller.
83       */
84      public String getReverseProxyAuthInitUrl() {
85  
86          return settings.getString(CoreProperties.SERVER_BASE_URL) + "/sessions/init/reverseproxyauth";
87      }
88  
89      /**
90       * @return URL that points to the unauthorized page.
91       */
92      public String getUnauthorizedUrl() {
93  
94          return settings.getString(CoreProperties.SERVER_BASE_URL) + "/sessions/unauthorized";
95      }
96  
97      /**
98       * Gets the user name from the headers of the request. This may return null,
99       * but will never return an empty or blank string.
100      *
101      * @param request
102      *            servlet request
103      * @return user name
104      */
105     public String getUserNameFromHeader(@Nonnull final HttpServletRequest request) {
106 
107         final String headerValue = request.getHeader(settings.getString(HEADER_NAME));
108         if (headerValue == null || headerValue.trim().isEmpty()) {
109             return null;
110         } else {
111             return headerValue;
112         }
113     }
114 
115     /**
116      * Checks if the server name is equal to localhost in the servlet request.
117      *
118      * @param request
119      *            servlet request
120      * @return true if the server name is equal to localhost.
121      */
122     public boolean isLocalHost(@Nonnull final ServletRequest request) {
123 
124         final String localhostValue = settings.getString(LOCALHOST);
125         return localhostValue != null && localhostValue.equals(request.getServerName());
126     }
127 
128     /**
129      * Returns <code>true</code> if the security realm is set to
130      * reverseproxyauth.
131      *
132      * @return <code>true</code> if the security realm is set to
133      *         reverseproxyauth.
134      */
135     public boolean isRealmReverseProxyAuth() {
136 
137         return ReverseProxyAuthPlugin.KEY.equals(settings.getString(CoreProperties.CORE_AUTHENTICATOR_REALM));
138     }
139 }
